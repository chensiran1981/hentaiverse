<script type="text/javascript">

var attrs = new Array("str","dex","agi","end","int","wis");
var total_exp = 518770067;
var usable_exp = 518770067;
var doovers = 10;

function get_total_expcost(stat_level) {
return stat_level == 0 ? 0 : Math.round(Math.pow(stat_level + 1, Math.pow(2.5475566751265, 1 + (stat_level / 950))));
}

function get_next_expcost(stat_level) {
return get_total_expcost(stat_level + 1) - get_total_expcost(stat_level);
}

function update_usable_exp() {
usable_exp = total_exp;
for(i in attrs) {
var current_level = get_pa(attrs[i]) + get_pa_delta(attrs[i]);
usable_exp -= get_total_expcost(current_level);
}
//common.hv_debug("Recalculated usable_exp=" + usable_exp);
}

function update_delta(which, stat_delta) {
document.getElementById(which + "_delta").value = stat_delta;
}

function update_display(which, stat_level, stat_delta) {
document.getElementById(which + "_display").innerHTML = common.get_dynamic_digit_string(stat_level + stat_delta);
document.getElementById(which + "_text").innerHTML = common.get_dynamic_digit_string(stat_delta);
document.getElementById("remaining_exp").innerHTML = common.get_dynamic_digit_string(usable_exp);

for(i in attrs) {
var current_level = get_pa(attrs[i]) + get_pa_delta(attrs[i]);
var next_exp = get_next_expcost(current_level);
var enable_inc = next_exp <= usable_exp;
var enable_dec = get_pa_delta(attrs[i]) > 0 || (doovers > 0 && current_level > 0);
document.getElementById(attrs[i] + "_inc").src = "/y/character/inc" + (enable_inc ? "" : "_d") + ".png";
document.getElementById(attrs[i] + "_dec").src = "/y/character/dec" + (enable_dec ? "" : "_d") + ".png";
document.getElementById(attrs[i] + "_left").innerHTML = common.get_dynamic_digit_string(next_exp);
}
}

function get_pa(which) {
return parseInt(document.getElementById(which).value);
}

function get_pa_delta(which) {
return parseInt(document.getElementById(which + "_delta").value);
}

function increase_pa(which, event) {
var increase_count = 1;

if( event.shiftKey ) {
increase_count = 10;
}
else if( event.ctrlKey ) {
increase_count = 100;
}
var level = get_pa(which);
var delta = get_pa_delta(which);

do {
if( get_next_expcost(level + delta) <= usable_exp ) {
if( delta < 0 ) {
doovers += 1;
}
update_delta(which, ++delta);
update_usable_exp();
}
else {
break;
}
} while( --increase_count > 0 );
update_display(which, level, delta);
}

function decrease_pa(which, event) {
var decrease_count = 1;

if( event.shiftKey ) {
decrease_count = 10;
}
else if( event.ctrlKey ) {
decrease_count = 100;
}
var level = get_pa(which);
var delta = get_pa_delta(which);

do {
if( delta > 0 || ((doovers > 0) && (level + delta > 0)) ) {
if( delta <= 0 ) {
doovers -= 1;
}
update_delta(which, --delta);
update_usable_exp();
}
else {
break;
}
} while( --decrease_count > 0 );

update_display(which, level, delta);
}

function do_persona_post() {
document.getElementById("personaform").submit();
}

function do_persona_rename() {
var name = prompt("Enter a new name for this persona.", "Default Persona");
if( name != null ) {
document.getElementById("persona_rename").value = name;
do_persona_post();
}
}

function do_persona_create_clone() {
if( confirm("Are you sure you wish to create a new persona with the same attribute, slot, equipment and ability assignments as \"Default Persona\"? This action is irreversible, and created personas cannot be deleted.") ) {
document.getElementById("persona_create").value = "clone";
do_persona_post();
}
}

function do_persona_create_blank() {
if( confirm("Are you sure you wish to create a blank persona? This action is irreversible, and created personas cannot be deleted.") ) {
document.getElementById("persona_create").value = "blank";
do_persona_post();
}
}
